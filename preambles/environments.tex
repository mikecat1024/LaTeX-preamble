\newcommand*{\SelfNumber}{?}
\NewDocumentCommand\environment{m m}{
  \newtheorem{_#1}[BaseCounter]{#2}
  \newtheorem*{_#1*}{#2}

  }{
    \end{_#1*}
  }
}
