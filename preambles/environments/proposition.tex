\newtheoremstyle{\PropositionCommand}{}{}{}{}{\bfseries}{.\\}{1zw}{\thmname{#1}\thmnumber{\hspace{0.2em}#2}\thmnote{\hspace{0.5em}(#3)}}
\theoremstyle{\PropositionCommand}
\newtheorem{_\PropositionCommand}[BaseCounter]{\PropositionCaption}
\newtheorem*{\PropositionCommand*}{\PropositionCaption}

\NewDocumentEnvironment{\PropositionCommand}{o d()}{
  \IfValueT{#2}{
    \setcounter{tmp}{\value{BaseCounter}}
    \setcounterref{BaseCounter}{#2}
    \addtocounter{BaseCounter}{-1}
  }
  \IfValueTF{#1}{\begin{_\PropositionCommand}[#1]}{\begin{_\PropositionCommand}}
  \IfValueT{#2}{
    \setcounter{BaseCounter}{\value{tmp}}
  }
}
{\end{_\PropositionCommand}}
